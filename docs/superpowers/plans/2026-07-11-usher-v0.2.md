# usher v0.2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship usher v0.2 — headless mode (`usher -p`) with automatic quota failover, GitHub Copilot CLI and Cursor CLI adapters, and the v0.1 review follow-ups.

**Architecture:** One new `Adapter` method (`HeadlessArgs`), two new one-file adapters, a `cmdHeadless` flow in main.go that loops the ranked pool instead of prompting, and ledger-recording moved so every actual launch is recorded. No new packages.

**Tech Stack:** unchanged — Go 1.26, stdlib + BurntSushi/toml only.

## Global Constraints

- Everything from v0.1 still binds: module `github.com/theodorebeaupre-prog/usher`; only dep BurntSushi/toml; `gofmt -l .` empty + `go vet ./...` clean per commit; commit trailer `Co-Authored-By: Claude Fable 5 <noreply@anthropic.com>`; 4-bit ANSI gated on `useColor()`; ledger never fails a launch; router stays pure.
- Agent name strings are now exactly: `claude`, `codex`, `gemini`, `opencode`, `copilot`, `cursor` — registry order exactly that (founding four, then alphabetical).
- In `-p` mode: stdout carries ONLY agent output; every usher line (decision, failover, errors) goes to stderr; the banner never appears.
- `HeadlessArgs(prompt)` returns nil when headless is unsupported OR prompt is empty; nil-returning agents are excluded from the `-p` pool before ranking.
- Forced agent (`--agent X -p`) = single attempt, no failover.
- Reality wins: verify every CLI flag against the installed CLI's `--help` before trusting this plan; fix code AND tests to match reality and record findings in the report.
- The current main.go was amended after the original v0.1 plan (color helpers `useColor()`/`sgr(...)`, `printDecision` with a routed-around parameter, `strongestCappedLoser`, stray-flag rejection). Read the actual file before editing; adapt integration points to what exists — real code wins over this plan's sketches.

```go
// cmdHeadless routes and runs the winner's print-and-exit mode. usher's own
// lines go to stderr; stdout belongs entirely to the agent. On a quota
// error it fails over to the next ranked agent automatically — each agent
// is attempted at most once.
func cmdHeadless(task, forced string, why bool) error {
	now := time.Now()
	cfg, err := config.Load(filepath.Join(config.Dir(), "config.toml"))
	if err != nil {
		fmt.Fprintln(os.Stderr, "usher: config error, using defaults:", err)
		cfg = config.Config{}
	}
	led := ledger.Load(filepath.Join(config.Dir(), "ledger.json"))
	if led.Warning != "" {
		fmt.Fprintln(os.Stderr, "usher:", led.Warning)
	}

	dir, _ := os.Getwd()
	var infos []router.AgentInfo
	for _, info := range installedAgents(led, now) {
		a, _ := adapters.Get(info.Name)
		if a.HeadlessArgs(task) == nil {
			continue // no print-and-exit mode; not a candidate for -p
		}
		infos = append(infos, info)
	}
	if len(infos) == 0 {
		return noHeadlessAgentsError()
	}
	ranked := router.Rank(task, dir, infos, cfg)
	if len(ranked) == 0 {
		return fmt.Errorf("all headless-capable agents are disabled in config")
	}

	if forced != "" {
		var pick *router.Scored
		for i := range ranked {
			if ranked[i].Name == forced {
				pick = &ranked[i]
				break
			}
		}
		if pick == nil {
			return fmt.Errorf("--agent %s: not installed, disabled, or no headless mode (try: usher doctor)", forced)
		}
		ranked = []router.Scored{*pick} // single attempt, no failover
	}

	if why {
		printWhy(ranked, ranked[0].Name)
	}

	exit := 0
	for i, choice := range ranked {
		a, _ := adapters.Get(choice.Name)
		fmt.Fprintf(os.Stderr, "%s %s%s\n",
			sgr(useColor(), 96, "→ "+choice.Name),
			sgr(useColor(), 90, "("+choice.TaskType+" task · headless)"),
			"")
		led.RecordLaunch(choice.Name, time.Now())
		if err := led.Save(); err != nil {
			fmt.Fprintln(os.Stderr, "usher: could not save ledger:", err)
		}
		var tail string
		exit, tail, err = launch.Run(a.Bin(), a.HeadlessArgs(task), dir)
		if err != nil {
			fmt.Fprintf(os.Stderr, "usher: %s failed to start: %v\n", choice.Name, err)
			exit = 1
			continue
		}
		if !a.QuotaError(exit, tail) {
			os.Exit(exit)
		}
		led.RecordQuota(choice.Name, time.Now())
		if err := led.Save(); err != nil {
			fmt.Fprintln(os.Stderr, "usher: could not save ledger:", err)
		}
		if i < len(ranked)-1 {
			fmt.Fprintf(os.Stderr, "%s\n", sgr(useColor(), 93,
				fmt.Sprintf("→ %s hit its cap — failing over to %s", choice.Name, ranked[i+1].Name)))
		}
	}
	fmt.Fprintln(os.Stderr, "usher: every available agent is capped or failed — exiting with the last agent's code")
	os.Exit(exit)
	return nil
}

func noHeadlessAgentsError() error {

	sb.WriteString("no installed agent supports headless mode. These would:\n")

			sb.WriteString(fmt.Sprintf("  %-10s %s\n", a.Name(), a.Profile().InstallHint))
		}
	}
	return fmt.Errorf("%s", sb.String())
}
```

```go
package tests

import (

// runSplit is like run but keeps stdout and stderr separate — headless
// stdout purity is part of the contract.
func runSplit(t *testing.T, bin, fakeDir, cfgHome string, args ...string) (stdout, stderr string, exit int) {
	t.Helper()
	cmd := exec.Command(bin, args...)
	cmd.Env = []string{"PATH=" + fakeDir, "HOME=" + cfgHome, "XDG_CONFIG_HOME=" + cfgHome}
	var out, errb strings.Builder
	cmd.Stdout, cmd.Stderr = &out, &errb
	_ = cmd.Run()
	return out.String(), errb.String(), cmd.ProcessState.ExitCode()
}

func TestHeadlessPassthroughAndExitCode(t *testing.T) {
	bin := buildUsher(t)
	fakeDir, cfgHome := t.TempDir(), t.TempDir()
	fakeAgent(t, fakeDir, "claude", `
if [ "$1" = "--version" ]; then echo "1.0.0 (fake)"; exit 0; fi
if [ "$1" = "-p" ]; then echo "HEADLESS_ANSWER: $2"; exit 0; fi
echo "unexpected: $@" >&2; exit 9`)

	stdout, stderr, exit := runSplit(t, bin, fakeDir, cfgHome, "-p", "fix the flaky auth test")
	if exit != 0 {
		t.Fatalf("exit = %d\nstderr: %s", exit, stderr)
	}
	if !strings.Contains(stdout, "HEADLESS_ANSWER: fix the flaky auth test") {
		t.Errorf("stdout missing agent output:\n%s", stdout)
	}
	if strings.Contains(stdout, "→") {
		t.Errorf("usher decision line leaked onto stdout:\n%s", stdout)
	}
	if !strings.Contains(stderr, "→ claude") {
		t.Errorf("decision line missing from stderr:\n%s", stderr)
	}
}

func TestHeadlessFailover(t *testing.T) {
	bin := buildUsher(t)
	fakeDir, cfgHome := t.TempDir(), t.TempDir()
	fakeAgent(t, fakeDir, "claude", `
if [ "$1" = "--version" ]; then echo "1.0.0 (fake)"; exit 0; fi
echo "usage limit reached" >&2; exit 1`)
	fakeAgent(t, fakeDir, "codex", `
if [ "$1" = "--version" ]; then echo "1.0.0 (fake)"; exit 0; fi
if [ "$1" = "exec" ]; then echo "CODEX_RESULT: $2"; exit 0; fi
echo "unexpected: $@" >&2; exit 9`)

	// "fix the crash" classifies as debug: claude (0.90) outranks codex (0.85),
	// so claude goes first, hits its fake cap, and usher must fail over.
	stdout, stderr, exit := runSplit(t, bin, fakeDir, cfgHome, "-p", "fix the crash")
	if exit != 0 {
		t.Fatalf("exit = %d\nstderr: %s", exit, stderr)
	}
	if !strings.Contains(stdout, "CODEX_RESULT: fix the crash") {
		t.Errorf("failover output missing:\n%s", stdout)
	}
	if !strings.Contains(stderr, "failing over to codex") {
		t.Errorf("failover notice missing from stderr:\n%s", stderr)
	}
	data, err := os.ReadFile(filepath.Join(cfgHome, "usher", "ledger.json"))
	if err != nil {
		t.Fatalf("ledger not written: %v", err)
	}
```
