```go
var ErrTimeout = errors.New("agent timed out")

func Run(bin string, args []string, dir string, opts Opts) (int, string, string, error) {
	path, err := exec.LookPath(bin)
	if err != nil {
		return -1, "", "", err
	}
	ctx := context.Background()
	if opts.Timeout > 0 {
		var cancel context.CancelFunc
		ctx, cancel = context.WithTimeout(ctx, opts.Timeout)
		defer cancel()
	}
	tail := NewTail(tailSize)
	var outBuf strings.Builder
	cmd := exec.CommandContext(ctx, path, args...)
	cmd.Dir = dir
	if opts.Stdin != nil {
		cmd.Stdin = opts.Stdin
	} else {
		cmd.Stdin = os.Stdin
	}
	if opts.CaptureStdout {
		cmd.Stdout = &outBuf
	} else {
		cmd.Stdout = os.Stdout
	}
	cmd.Stderr = io.MultiWriter(os.Stderr, tail)

	signal.Ignore(os.Interrupt)
	defer signal.Reset(os.Interrupt)

	err = cmd.Run()
	if ctx.Err() == context.DeadlineExceeded {
		return 124, outBuf.String(), tail.String(), ErrTimeout
	}
	var exitErr *exec.ExitError
	if err != nil && !errors.As(err, &exitErr) {
		return -1, outBuf.String(), tail.String(), err
	}
	return cmd.ProcessState.ExitCode(), outBuf.String(), tail.String(), nil
}
```

### Task 5: --json envelope

**Files:** Modify `main.go`, `tests/e2e_headless_test.go`

**Interfaces — envelope (exact):**

```go
type jsonAttempt struct {
	Agent      string `json:"agent"`
	ExitCode   int    `json:"exit_code"`
	QuotaError bool   `json:"quota_error"`
	TimedOut   bool   `json:"timed_out"`
	DurationMS int64  `json:"duration_ms"`
}
type jsonEnvelope struct {
	Agent      string        `json:"agent"`
	TaskType   string        `json:"task_type"`
	ExitCode   int           `json:"exit_code"`
	DurationMS int64         `json:"duration_ms"`
	Output     string        `json:"output"`
	Attempts   []jsonAttempt `json:"attempts"`
}
```

- [ ] **Step 1:** flag `jsonFlag := fs.Bool("json", false, "headless: wrap the result in a JSON envelope on stdout")`; without `-p` → error `--json requires -p (headless mode)`. Thread into `cmdHeadless(task, forced string, why bool, timeout time.Duration, asJSON bool)`.
- [ ] **Step 2:** in the attempt loop: `opts.CaptureStdout = asJSON`; time each attempt (`time.Since`); append a `jsonAttempt` per attempt (incl. timeout/start-fail ones — start-fail: exit -1). Where the non-JSON path calls `os.Exit(exit)`, the JSON path first marshals and prints the envelope (final attempt's agent/exit/output; total duration = sum of attempts) then exits with the same code. Timeout path in JSON mode also emits the envelope (TimedOut true, exit 124) before `os.Exit(124)`. Factor a small `emitJSON(env)` helper; keep stdout pure — the envelope is the only stdout write (`CaptureStdout` guarantees the agent never touched it).
